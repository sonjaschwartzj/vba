Private Sub UserForm_Initialize()
Dim cell As Range
With Me.cboAktion
    .AddItem "Auftrag"
    .AddItem "LS"
    .AddItem "Rechnung"
    .AddItem "Abo Rechnung"
    .AddItem "ICE Rechnung"
End With
For Each cell In Range("A2:A1000000")
    If cell <> "" Then
        cboAnNr.AddItem cell
    Else
    Exit For
    End If
Next cell
End Sub
